        function analyzeInputs() {

            const reason1 = document.getElementById("reason1").value;

            const reason2 = document.getElementById("reason2").value;

            const reason3 = document.getElementById("reason3").value;

            const allReasons = reason1 + " " + reason2 + " " + reason3;


            const characteristics = [];

            const emotions = [];

            const motivationSummary = [];


            // 簡単なキーワードマッチングによる特性と感情の抽出

            const keywords = {

                "挑戦": { characteristic: "挑戦心", emotion: "意欲" },

                "目標達成": { characteristic: "目標達成意欲", emotion: "達成感" },

                "協力": { characteristic: "協調性", emotion: "連帯感" },

                "学び": { characteristic: "探求心", emotion: "好奇心" },

                "成長": { characteristic: "成長意欲", emotion: "向上心" },

                "困難": { characteristic: "忍耐力", emotion: "不屈の精神" },

                "工夫": { characteristic: "創造性", emotion: "発見" },

                "貢献": { characteristic: "貢献意欲", emotion: "喜び" },

                "責任": { characteristic: "責任感", emotion: "使命感" },

                "リーダーシップ": { characteristic: "リーダーシップ", emotion: "鼓舞" }

                // 必要に応じてキーワードを追加・修正してください

            };


            for (const word in keywords) {

                if (allReasons.includes(word)) {

                    if (!characteristics.includes(keywords[word].characteristic)) {

                        characteristics.push(keywords[word].characteristic);

                    }

                    if (!emotions.includes(keywords[word].emotion)) {

                        emotions.push(keywords[word].emotion);

                    }

                }

            }


            document.getElementById("characteristics-list").innerText = characteristics.join("、");

            document.getElementById("emotions-list").innerText = emotions.join("、");


            // 簡易的なモチベーションの傾向生成

            if (emotions.length > 0) {

                motivationSummary.push("これらの経験から、");

                motivationSummary.push(emotions.join("、") + "といった感情がモチベーションの源泉となっていると考えられます。");

                if (characteristics.includes("挑戦心") || characteristics.includes("目標達成意欲")) {

                    motivationSummary.push("目標達成に向けて積極的に行動する傾向が見られます。");

                }

                if (characteristics.includes("協調性") || characteristics.includes("貢献意欲")) {

                    motivationSummary.push("チームや周囲との連携を重視し、貢献することに喜びを感じるかもしれません。");

                }

                // より詳細な分析に基づいて条件分岐を追加することも可能です

                document.getElementById("motivation-graph").innerText = motivationSummary.join("");

            } else {

                document.getElementById("motivation-graph").innerText = "記述内容からモチベーションの傾向を分析中です。";

            }

        }

    </script>
